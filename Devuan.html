Devuan

Devuan is a GNU/Linux operating system and fork of the Debian distribution. It is best choice for developers because it does not require complexity of systemd and dbus components.

  • https://devuan.org/

CD Image

The listed scripts create a new root filesystem with debootstrap program. It does not use variables only file descriptors and pipes. File descriptors connect the top level script with the lower level schroot/bash shells. The generated files are available in the cdimage directory.

  • https://lippydanger.jumpingcrab.com/cdimage/

        -t overlay \

cd debian || exit 1
bash -e <<-'EOF'
        # read packages with filename
        < /var/lib/apt/lists/deb.devuan.org_merged_dists_testing_main_binary-amd64_Packages \
        grep ^Package\\\|^Priority\\\|^Filename |
        paste - - - |
        sort |
        nl -n ln |
        tee >(
                # join required and important packages
                sort -k5 |
                join -15 - <(
                        cat <<<$'important\nrequired'
                ) |
                cut -d\  -f4,7 |
                nl -n ln |
                join -j5 <(seq 1) -
        ) > >(
                # join desired packages
                sort -k3 |
                join -13 - <(
                        # append packages and dependencies
                        <&3 tee >(
                                xargs apt-cache depends \
                                        --recurse \
                                        --installed \
                                        --no-pre-depends \
                                        --no-suggests \
                                        --no-recommends \
                                        --no-conflicts \
                                        --no-breaks \
                                        --no-replaces \
                                        --no-enhances |
                                grep Depends: |
                                cut -d: -f2 |
                                cut -d\  -f2 |
                                grep -v '^<'
                        ) | sort | uniq
                ) |
                cut -d\  -f1,7 |
                sort -n -k2 |
                nl -n ln |
                join -j5 <(seq 2 2) -
        ) |
        cut -d\  -f4,5 |
        sort | uniq |

        # format input for shell reuse
        grep -o \\S* |
        xargs -I@ printf %q\\n '@' |
        sed -e 1~2s/^/basename\ /e -e 0~2s/^/dirname\ /e |
        xargs -I@ printf %q\\n '@' |

        # divide package list into directories
        paste - - |
        nl -n ln |
        tee >(
                # package names
                cut -f1,2 | join <(seq 2 2) - -j3
        ) > >(
                # uniq directory names
                cut -f1,3 | join <(seq 1) - -j3 | uniq -i2
        ) |
        sort -k2 -n |
        cut -d\  -f2,4 |

        # group packages by folders in one line
        sed /^1\ /i\\\\\x00 |
        tail -z -n+2 |
        cut -d\  -f2 |
        tr \\n\\\000 \\\ \\\n |

        # print apt commands
        sed -f<(cat <<-'SED'
                s/^\s*\([[:graph:]]*\)\s*\(.*\)$/\
                mkdir -p \1 \&\& cd \1 \&\& apt-get download \2/
        SED
        ) |

        # execute parallel apt-get
        xargs --replace=@ --max-procs=$(nproc) sh -c '@'

        # rename packages with epochs version
        find -name '*%*' -exec dpkg-name -o {} +

        # transition
        # find -name libcomerr2_1.43.4-2_amd64.deb -delete

        #cat && exit 1

        mkdir -p pool/main/d
        bash -e <<-'BASH'
                cd pool/main/d
                xargs apt-get source --download-only <<-XARGS
                        debian-installer
                        debian-keyring
                        debootstrap
                        devuan-baseconf
                        devuan-keyring
                XARGS
        BASH

        # download debian installer in debian/dists
        : wget --continue \
                http://deb.devuan.org/devuan/pool/main/d/debian-installer/debian-installer_20180717+devuan1{.dsc,.tar.gz,_amd64.deb}
